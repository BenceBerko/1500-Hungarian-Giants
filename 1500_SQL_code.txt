    COUNT(*) AS cegek_szama,

SELECT 
    "Varmegye",
    "Tevekenyseg",
    SUM("net_arbevetele") AS osszes_arbevetel,
    SUM("eredmeny") AS osszes_eredmeny,
    CASE 
        WHEN SUM("net_arbevetele") = 0 THEN 0 
        ELSE ROUND(
            (SUM("eredmeny")::float / NULLIF(SUM("net_arbevetele"), 0)::float * 100)::numeric, 4
        )
    END AS profit_marzs_szazalek
FROM 
    "1500"."Top1500_Megye"
GROUP BY 
    "Varmegye", "Tevekenyseg"
ORDER BY 
    profit_marzs_szazalek DESC;

--Legnagyobb árbevételt termelő tevékenység / Vármegye 

WITH RangsoroltArbevetel AS (
    SELECT 
        "Varmegye", 
        "Tevekenyseg",
        SUM(CAST(REPLACE("net_arbevetele"::text, ' ', '') AS BIGINT)) AS osszes_arbevetel,
        DENSE_RANK() OVER (
            PARTITION BY "Varmegye" 
            ORDER BY SUM(CAST(REPLACE("net_arbevetele"::text, ' ', '') AS BIGINT)) DESC
        ) as rangsor
    FROM 
        "1500"."Top1500_Megye"
    GROUP BY 
        "Varmegye", 
        "Tevekenyseg"
)
SELECT 
    "Varmegye", 
    "Tevekenyseg", 
    osszes_arbevetel
FROM 
    RangsoroltArbevetel
    WHERE 
    rangsor <= 5
ORDER BY 
    "Varmegye" ASC, 
    osszes_arbevetel DESC;

--Legnagyobb árbevételt termelő tevékenység / 1 db kiválasztott Vármegye 

WITH RangsoroltArbevetel AS (
    SELECT 
        "Varmegye", 
        "Tevekenyseg",
        SUM(CAST(REPLACE("net_arbevetele"::text, ' ', '') AS BIGINT)) AS osszes_arbevetel,
        DENSE_RANK() OVER (
            ORDER BY SUM(CAST(REPLACE("net_arbevetele"::text, ' ', '') AS BIGINT)) DESC
        ) as rangsor
    FROM 
        "1500"."Top1500_Megye"
    WHERE 
        "Varmegye" = 'Nógrád' -- Itt szűrünk rá Nógrád megyére
    GROUP BY 
        "Varmegye", 
        "Tevekenyseg"
)
SELECT 
    "Varmegye", 
    "Tevekenyseg", 
    osszes_arbevetel
FROM 
    RangsoroltArbevetel
WHERE 
    rangsor <= 5
ORDER BY 
    osszes_arbevetel DESC;

SELECT 
    t1500."Varmegye",
    COUNT(t1500."Cegnev") AS cegek_szama,
    SUM(CAST(REPLACE(t1500."net_arbevetele"::text, ' ', '') AS BIGINT)) AS osszes_arbevetel,
    SUM(CAST(REPLACE(t1500."eredmeny"::text, ' ', '') AS BIGINT)) AS osszes_eredmeny,
    CASE 
        WHEN SUM(CAST(REPLACE(t1500."net_arbevetele"::text, ' ', '') AS BIGINT)) = 0 THEN 0 
        ELSE ROUND(
            (SUM(CAST(REPLACE(t1500."eredmeny"::text, ' ', '') AS BIGINT))::float / 
             SUM(CAST(REPLACE(t1500."net_arbevetele"::text, ' ', '') AS BIGINT)) * 100)::numeric, 2)
    END AS profit_marzs,
    MAX(s."SUM2024") AS megyei_nepesseg,
    SUM(CAST(REPLACE(t1500."Lakasok_szama"::text, ' ', '') AS BIGINT)) AS megyei_lakas,
    CASE 
        WHEN MAX(s."SUM2024") = 0 OR MAX(s."SUM2024") IS NULL THEN 0 
        ELSE ROUND((SUM(CAST(REPLACE(t1500."eredmeny"::text, ' ', '') AS BIGINT))::float / 
             MAX(s."SUM2024"))::numeric, 0)
    END AS profit_per_lakos,
    CASE 
        WHEN MAX(CAST(REPLACE(t1500."Lakasok_szama"::text, ' ', '') AS BIGINT)) = 0 THEN 0 
        ELSE ROUND((SUM(CAST(REPLACE(t1500."eredmeny"::text, ' ', '') AS BIGINT))::float / 
             MAX(CAST(REPLACE(t1500."Lakasok_szama"::text, ' ', '') AS BIGINT)))::numeric, 0)
    END AS profit_per_lakas
FROM 
    "1500"."Top1500_Megye" AS t1500
LEFT JOIN 
    "1500"."stadat-nepessegFNO-hu" AS s ON t1500."Varmegye" = s."Varmegye"
GROUP BY 
    t1500."Varmegye"
ORDER BY 
    profit_marzs DESC;

        WHEN RIGHT("Adószám", 2) IN ('14', '34') THEN 'Somogy'
        WHEN RIGHT("Adószám", 2) IN ('15', '35') THEN 'Szabolcs-Szatmár-Bereg'

        WHEN RIGHT("Adószám", 2) IN ('17', '37') THEN 'Tolna'
        WHEN RIGHT("Adószám", 2) IN ('18', '38') THEN 'Vas'
        WHEN RIGHT("Adószám", 2) IN ('19', '39') THEN 'Veszprém'
        WHEN RIGHT("Adószám", 2) IN ('20', '40') THEN 'Zala'
        WHEN RIGHT("Adószám", 2) = '41' THEN 'Észak-Budapest'
        WHEN RIGHT("Adószám", 2) = '42' THEN 'Kelet-Budapest'
        WHEN RIGHT("Adószám", 2) = '43' THEN 'Dél-Budapest'
        WHEN RIGHT("Adószám", 2) = '44' THEN 'Kiemelt Adózók Adóigazgatósága'

    "1500"."1500sheet"

    1
ORDER BY 
    "Össz árbevétel (eFt)" DESC;
